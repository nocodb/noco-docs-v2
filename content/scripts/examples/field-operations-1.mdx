```javascript
// NocoDB Script: Create Fields of All Major Types
// This script creates a table and adds example fields for a wide range of field types using string literal types.

// Additional details about various options supported can be found here
// https://nocodb.com/apis/v3/meta#tag/Fields

const tableName = 'Field Types Demo';
let table = base.getTable(tableName);

// Create table if it doesn't exist
if (!table) {
  table = await base.createTableAsync(tableName, []);
  output.text(`✅ Created table: "${tableName}"`);
} else {
  output.text(`ℹ️ Table "${tableName}" already exists.`);
}

// Utility to create field only if missing
async function createField(name, config) {
  if (!table.getField(name)) {
    await table.createFieldAsync({ name, ...config });
    output.text(`✅ Created field: "${name}" (${config.type})`);
  } else {
    output.text(`ℹ️ Field "${name}" already exists.`);
  }
}

// create field with default value & description
await createField('newField', {
  type: "SingleLineText",
  default_value: "Default value for text field",
  description: "Demo field for default value & description"
})

// Text based
//
await createField('Name', { type: "SingleLineText" });
await createField('Description', { type: "LongText" });
await createField('Phone', { type: "PhoneNumber" });
await createField('Email', { type: "Email" });
await createField('Website', { type: "URL" });

// Numerical
//
await createField('Amount', { type: "Number" });
await createField('Budget', { type: "Currency" });
await createField('Conversion Rate', { type: "Decimal" });
await createField('Progress %', { type: "Percent" });

// Select
//
await createField('Tags', {
  type: "MultiSelect",
  options: {
    choices: [
      { title: 'UI', color: '#FFC107' },
      { title: 'API', color: '#28A745' }
    ]
  }
});
await createField('Status', {
  type: "SingleSelect",
  options: {
    choices: [
      { title: 'Draft', color: '#6c757d' },
      { title: 'Published', color: '#007BFF' }
    ]
  }
});

// Date & time
//
await createField('Start Date', { type: "Date" });
await createField('Start Time', { type: "Time" });
await createField('Due DateTime', { type: "DateTime" });
// await createField('Year Started', { type: "Year" });
await createField('Duration (mins)', {
  type: "Duration",
  options: { duration_format: "h:mm" }
});

// System
//
await createField('Created By', { type: "CreatedBy" });
await createField('Last Modified By', { type: "LastModifiedBy" });
await createField('Created Time', { type: "CreatedTime" });
await createField('Last Modified Time', { type: "LastModifiedTime" });

// Virtual
//
const nameField = table.getField("Name");
await createField('Barcode', {
  type: "Barcode",
  options: { barcode_value_field_id: nameField.id }
});
// await createField('QR Code', {
//   type: "QrCode",
//   options: { qrcode_value_field_id: nameField.id }
// });
await createField('Formula', {
  type: "Formula",
  options: { formula: 'CONCAT("Username is ", {Name})' }
});

// User
//
await createField('User Picker', { type: "User" });

// Others
//
await createField('Attachment', { type: "Attachment" });
await createField('Is Published', {
  type: "Checkbox",
  options: { icon: "square" }
});
await createField('Priority', {
  type: "Rating",
  options: { icon: 'star', max_value: 5, color: '#F4B400' }
});
await createField('Extra Info', { type: "JSON" });
await createField('Geometry Data', { type: "Geometry" });

output.markdown('### 📋 Final Field List');

output.table(table.fields.map(f => ({
  'Field Name': f.name,
  'Type': f.type,
  'Field ID': f.id
})));

```
